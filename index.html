helper <- function(x) rnorm(x)
fun <- function(x) helper(x)



library("parallel")
cl <- makeCluster(2)
clusterMap(cl, fun, 1:2)
stopCluster(cl)

One solution

library("modules")
m <- module({
  
  import("stats", "rnorm")
  
  helper <- function(x) rnorm(x)
  fun <- function(x) helper(x)
  
})

library("parallel")
cl <- makeCluster(2)
clusterMap(cl, m$fun, 1:2)
stopCluster(cl)

Use case: functions

Function with side-effect

plotFun <- function(data, ...) {
  
  library("ggplot2")
  
  ggplot(data, aes(...)) + 
    geom_boxplot() +
    ...
  
}

Alternative

plotFun <- function(data, ...) {
  
  modules::import("ggplot2")
  
  ggplot(data, aes(...)) + 
    geom_boxplot() +
    ...
  
}
